98 std::ostream & NodeXML::toOStr(std::ostream &ostr, const drain::Tree<K,T,C> & tree, const std::string & defaultTag){
99 
100  const std::map<std::string, Tree<K,T,C> > & children = tree.getChildren();
101 
102  // OPEN TAG
103  ostr << '<';
104  //<< tree.data.tag << ' ';
105  if (tree->getTag().empty())
106  ostr << defaultTag << ' ';
107  else {
108  ostr << tree->getTag() << ' ';
109  // TODO if (tree.data.name.empty()) ?
110  if (!defaultTag.empty())
111  ostr << "name=\"" << defaultTag << '"' << ' ';
112  }
113 
114  if (tree.data.id >= 0)
115  ostr << "id=\"" << tree.data.id << '"' << ' ';
116 
118  for (ReferenceMap::const_iterator it = tree->begin(); it != tree->end(); it++){
119 
120  std::stringstream sstr;
121  sstr << it->second;
122  if (!sstr.str().empty()){
123  ostr << it->first << "=\"" << it->second << '"' << ' ';
124  //ostr << "test=\"" << sstr.str() << '"' << ' ';
125  }
126  }
127 
128  if ((children.size() == 0) && tree->ctext.empty() ){ // OR no ctext!
129  // close TAG
130  ostr << '/' << '>';
131  ostr << '\n';
132  }
133  else {
134  // close starting TAG
135  ostr << '>';
136  if (tree->ctext.empty())
137  ostr << '\n'; // TODO nextline
138  else
139  ostr << tree->ctext;
140 
142  for (typename std::map<std::string, Tree<K,T,C> >::const_iterator it = children.begin(); it != children.end(); it++){
143  toOStr(ostr, it->second, it->first);
144  //ostr << *it;
145  }
146  // add end </TAG>
147  ostr << '<' << '/' << tree->getTag() << '>';
148  ostr << '\n'; // TODO nextline
149 
150  if (tree.data.id >= 0)
151  ostr << "<!-- " << tree.data.id << " /-->\n";
152  }
153  return ostr;
154 }
