165  inline
166  void setPosition(int i, int j) const {
167 
168  //static const double DEG2RAD = M_PI/180.0;
169  //static const double RAD2DEG = 180.0/M_PI;
170 
171  this->current_i = i;
172  this->current_j = j;
173  current_range = odim.getBinDistance(i);
174  current_azm = static_cast<double>(j) * J2AZMDEG; // odim.getAzm
175 
176  current_sin = sin(current_azm * drain::DEG2RAD);
177  current_cos = cos(current_azm * drain::DEG2RAD);
178 
179  x = current_range*current_sin;
180  y = current_range*current_cos;
181  proj.projectFwd(current_range*current_sin, current_range*current_cos, lon, lat);
182  lon *= drain::RAD2DEG;
183  lat *= drain::RAD2DEG;
184 
185  }
