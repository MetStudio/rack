170  inline
171  void setPosition(int i, int j) const {
172 
173  this->current_i = i;
174  this->current_j = j;
175  current_range = odim.getBinDistance(i);
176  current_azm = static_cast<double>(j) * J2AZMDEG; // odim.getAzm
177 
178  current_sin = sin(current_azm * drain::DEG2RAD);
179  current_cos = cos(current_azm * drain::DEG2RAD);
180 
181  x = current_range*current_sin;
182  y = current_range*current_cos;
183  proj.projectFwd(current_range*current_sin, current_range*current_cos, lon, lat);
184  lon *= drain::RAD2DEG;
185  lat *= drain::RAD2DEG;
186 
187  }
