318  void probe4(int i, int j) {
319 
320  if (!isValidPos(i,j))
321  return;
322 
323  visit(i,j); // mark & update
324 
325  const int i0 = i;
326  const int j0 = j;

329  int iEnd;
330  while (move(i,j, 1,0)){
331  visit(i,j);
332  }
333  iEnd = i;
334  // Now i2 is the maximum valid i coordinate.
335 
336  // Rewrite code: iMax should support wrapping, so 200...256...10.
337 
339  i=i0;
340  j=j0;
341  while (move(i,j, -1,0)){
342  visit(i, j);
343  }
344  // Now i is the minimum valid i coordinate.
345 
346 
348  int i2, j2;
349  bool done = false;
350  //while (i != iEnd){
351  while (!done){
352  done = (i == iEnd);
353  i2 = i;
354  j2 = j-1;
355  if (handler.validate(i2, j2))
356  if (isValidMove(i,j, i2,j2))
357  probe4(i2,j2);
358  i2 = i;
359  j2 = j+1;
