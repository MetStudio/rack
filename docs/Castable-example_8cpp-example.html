using namespace drain;
template <class S, class T>
void demo(const T & value, char outputSeparator = ','){
std::cout << "Convert: " << drain::Type::getTypeChar(typeid(T)) << '>' << drain::Type::getTypeChar(typeid(S)) << '\n';
//std::cout << "Input type: " << drain::Type::getTypeChar(typeid(T)) << '\n';
std::cout << "Input: " << typeid(T).name() << '\n';
std::cout << "Storage: " << typeid(S).name() << '\n';
S data;
drain::Castable c(data);
c.setSeparator(outputSeparator);
std::cout << "Castable: ";
c = value;
c.toJSON(std::cout);
std::cout << '\n';
std::cout << "Referencer: ";
drain::Referencer r;
r.setSeparator(outputSeparator);
r.link(data);
r = value;
r.toJSON(std::cout);
std::cout << '\n';
std::cout << "Variable: ";
drain::Variable v;
v.setSeparator(outputSeparator);
v.setType(typeid(S));
v = value;
v.toJSON(std::cout);
std::cout << '\n';
drain::FlexVariable fv;
std::cout << "FlexVar, copied: ";
fv = data;
fv.toJSON(std::cout);
std::cout << '\n';
std::cout << "FlexVar, linked: ";
fv.link(data);
fv.toJSON(std::cout);
std::cout << '\n';
std::cout << "Var=Cas: ";
v.reset();
v = c;
v.toJSON(std::cout);

int main(int argc, char **argv){
if (argc <= 1){
std::cout << "Demonstrates storing values to Castables and Referencers.\n";
std::cout << "Usage:\n " << argv[0] << " <input1> <input2> ... ]\n";
std::cout << "Example:\n " << argv[0] << " 123.45 456.789E+6 Åland ... ]\n\n";
}
Range<double> reijo;
for (int i=1; i<argc; ++i){
const char *s = argv[i];
std::cout << "Arg " << i << ":\t" << s << '\n';
std::string str = s;
int k = atoi(s);
double d = atof(s);
//std::cout << "Storage type: string\n";
std::cout << "---------\n";
demo<std::string>(s);
demo<std::string>(k);
demo<std::string>(d);
std::cout << "---------\n";
demo<char>(s,0);
demo<char>(k,0);
demo<char>(d,0);
//std::cout << "Storage type: int\n";
std::cout << "---------\n";
demo<int>(s);
demo<int>(k);
demo<int>(d);
//std::cout << "Storage type: double\n";
std::cout << "---------\n";
demo<double>(s);
demo<double>(k);
demo<double>(d);
}
std::cout << "---------\n\n";
std::cout << "Storage type: vector<double>\n";
std::vector<double> v(3); v[0]=1; v[1]=2; v[2]=3;
Castable c(v);
c.toJSON(std::cout, '\n');
c = std::vector<unsigned int>(2, 1234);
c.toJSON(std::cout, '\n');
Referencer r;
r.link(v);
r = std::vector<double>(4, 678);
r.toJSON(std::cout, '\n');
c = "mika,0.2,712.122,192";

r.link(c);

//std::pair<int,int> range(12,34);

std::cout << '\n';
cr = 123;
cr << 5 << 6;

return 0;
}
