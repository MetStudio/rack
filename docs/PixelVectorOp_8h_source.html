174 void BinaryPixelVectorOp<F>::traverseChannels(const ImageTray<const Channel> & src1, const ImageTray<const Channel> & src2,
175  ImageTray<Channel> & dst) const {
176 
177 
178  Logger mout(getImgLog(), name+"(PixelVectorOp)", __FUNCTION__);
179 
180  if (src1.empty())
181  mout.error() << "src1 empty" << mout.endl;
182 
183  if (src2.empty())
184  mout.error() << "src2 empty" << mout.endl;
185 
186  if (dst.empty())
187  mout.error() << "dst empty" << mout.endl;
188 
189  Channel & dstChannel = dst.get();
190 
191  if (src1.getGeometry() != src2.getGeometry()){
192  mout.warn() << "src1" << src1.getGeometry() << mout.endl;
193  mout.warn() << "src2" << src2.getGeometry() << mout.endl;
194  mout.note() << "dstChannel" << dstChannel << mout.endl;
195  mout.error() << "src1 and src2 geometry error" << mout.endl;
196  return;
197  }
198 
199  const int width = src1.getGeometry().getWidth(); //std::min(src1.getWidth(),src2.getWidth());
200  const int height = src1.getGeometry().getHeight(); //std::min(src1.getHeight(),src2.getHeight());
201  const size_t channels = src1.size(); // std::min(src1.getChannelCount(),src2.getChannelCount());
202 
203  double x=0.0, sum=0.0;
204 
205  mout.debug(2) << *this << mout.endl;
206  //mout.debug(2) << "src1: " << src1 << mout.endl;
207  //mout.debug(2) << "src2: " << src2 << mout.endl;
208  //mout.debug(2) << "dst: " << dst << mout.endl;
209 
210  FunctorBank & functorBank = getFunctorBank();
211 
212  if (!functorBank.has(functorName)){
213  //functorBank.help(std::cerr);
214  mout.error() << "functor not found: " << functorName << mout.endl;
215  return;
216  }
217 
218  UnaryFunctor & scalingFunctor = functorBank.clone(functorName); // consider get? Or static from clone()
219  //mout.debug() << scalingFunctor.getName() << ':' << scalingFunctor.getDescription() << mout.endl;
220  //scalingFunctor.setScale(dstChannel.scaling.getMax<double>(), 0.0);
221  const std::type_info & t = dstChannel.getType();
222  if (Type::call<drain::typeIsSmallInt>(t)){
223  const double m = Type::call<typeMax, double>(t);
224  scalingFunctor.setScale(m, 0.0); // TODO consider invScale, invOffset?
225  mout.info() << "small int type, scaling with its maximum (" << m << ')' << mout.endl;
226  }
227 
228  scalingFunctor.setParameters(functorParams);
229  mout.debug(1) << scalingFunctor.getName() << ':' << scalingFunctor << mout.endl;
230 
231  const double coeff = (rescale>0.0) ? 1.0/rescale : 1.0/static_cast<double>(channels);
232  const bool USE_POW = (POW != 1.0);
233  const bool USE_INVPOW = (INVPOW != 1.0);
234  mout.debug(2) << "coeff " << coeff << mout.endl;
235  const BinaryFunctor & f = binaryFunctor;
236  size_t a;
237  for (int j = 0; j < height; j++) {
238  for (int i = 0; i < width; i++) {
239 
240  sum = 0.0;
241  a = dstChannel.address(i,j);
242  for (size_t k = 0; k < channels; k++){
243  // getValue( src1.get(k).get<double>(a) , src2.get(k).get<double>(a) );
244  x = f(src1.get(k).get<double>(a) , src2.get(k).get<double>(a)) ;
245  if (USE_POW)
246  sum += pow(x, POW);
247  else
248  sum += x;
249  }
250  if (USE_INVPOW)
251  dstChannel.put(a, scalingFunctor(pow(coeff*sum, INVPOW)));
252  else
253  dstChannel.put(a, scalingFunctor(coeff*sum));
254  }
255  }
