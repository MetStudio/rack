309  inline virtual
310  void m2pix(double x, double y, int & i, int & j) const {
311  i = static_cast<int>(0.5+ (x - extentNative.lowerLeft.x) / xScale); // xOffset
312  //j = frameHeight-1 - static_cast<int>(0.5+ (y - extentNative.lowerLeft.y) / yScale);
313  j = frameHeight-1 - static_cast<int>(0.5+ (y - extentNative.lowerLeft.y) / yScale);
314  }
315 
316  inline virtual

318  pImage.x = static_cast<int>(0.5+ (pMetric.x - extentNative.lowerLeft.x) / xScale); // xOffset
319  //pImage.y = frameHeight-1 - static_cast<int>(0.5+ (pMetric.y - extentNative.lowerLeft.y) / yScale);
320  pImage.y = frameHeight-1 - static_cast<int>(0.5+ (pMetric.y - extentNative.lowerLeft.y) / yScale);
321 
322  }
