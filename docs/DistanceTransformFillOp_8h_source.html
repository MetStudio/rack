226 void DistanceTransformFillOp<T>::traverseDownRight(const ImageTray<const Channel> & srcTray, ImageTray<Channel> & dstTray) const {
227 
228  Logger mout(getImgLog(), __FUNCTION__, __FILE__);
229 
230  mout.debug() << "distModel: " << this->distanceModel << mout;
231 
232  DistanceNeighbourhood chain;
233  this->distanceModel.createChain(chain, true);
234 
235  mout.debug2() << "neighbourHood " << drain::sprinter(chain) << mout;
236 
237  const Channel & srcAlpha = srcTray.getAlpha();
238  Channel & dstAlpha = dstTray.getAlpha();
239 
240  CoordinateHandler2D coordinateHandler(srcTray.get(0));
241  mout.debug() << "coordHandler " << coordinateHandler << mout;
242 
243  mout.debug3() << "src alpha:" << srcAlpha << mout;
244  mout.debug3() << "dst alpha:" << dstAlpha << mout;
245 
247  Point2D<int> p;
248  Point2D<int> pTest;
249  Point2D<int> pWin;
250 
251  // Winning distance
252  dist_t dWin;
253  // Distance candidate
254  dist_t dTest;
255 
256  const size_t K = std::min(srcTray.size(), dstTray.size());
257 
258  const Range<int> & xRange = coordinateHandler.getXRange();
259  const Range<int> & yRange = coordinateHandler.getYRange();
260 
261  size_t address = 0;
262  size_t addressWin = 0;
263 
264  mout.debug() << "main loop, K=" << K << mout.endl;
265  for (p.y=0; p.y<=yRange.max; ++p.y){
266  for (p.x=0; p.x<=xRange.max; ++p.x){
267 
268  address = dstAlpha.address(p.x, p.y);
269 
270  // Take source value as default
271  dWin = srcAlpha.get<dist_t>(address);
272  pWin.setLocation(p);
273 
274  for (const DistanceElement & elem: chain){
275  pTest.setLocation(p.x + elem.diff.x, p.y + elem.diff.y);
276  coordinateHandler.handle(pTest);
277  dTest = this->distanceModel.decrease(dstAlpha.get<dist_t>(pTest), elem.coeff);
278  if (dTest > dWin){
279  dWin = dTest;
280  pWin = pTest;
281  }
282  }
283 
284 
285  // Finally, update target image (if data found)
286  if (dWin > 0.0){
287 
288  dstAlpha.put(address, dWin);
289 
290  addressWin = dstAlpha.address(pWin.x, pWin.y);
291 
292  if (addressWin != address){
293  for (size_t k=0; k<K; ++k)
294  dstTray.get(k).put(address, dstTray.get(k).get<dist_t>(addressWin));
295  }
296  else {
297  for (size_t k=0; k<K; ++k)
298  dstTray.get(k).put(address, srcTray.get(k).get<dist_t>(address));
299  }
300 
301  }
302 
303  /*
304  if ((p.x == p.y) && ((p.x&15)==0)){
305  std::cerr << p << "\t a=" << address << ", aW=" << addressWin << "\t" << dWin << '>' << dTest << '\t';
306  std::cerr << p << "\t s=" << srcAlpha.get<dist_t>(address) << '=' << srcAlpha.get<dist_t>(p) << '\t';
307  std::cerr << p << "\t d=" << dstAlpha.get<dist_t>(address) << '=' << dstAlpha.get<dist_t>(p) << '\n';
308  }
309  */
310 
311  }
312  }

324 void DistanceTransformFillOp<T>::traverseUpLeft(ImageTray<Channel> & srcTray, ImageTray<Channel> & dstTray) const {
325 
326  Logger mout(getImgLog(), __FUNCTION__, __FILE__);
327  mout.debug() << "start" << mout.endl;
328 
329  mout.debug2() << "this->distanceModel" << this->distanceModel << mout.endl;
330 
331  DistanceNeighbourhood chain;
332  this->distanceModel.createChain(chain, false);
333 
334  CoordinateHandler2D coordinateHandler(srcTray.get(0));
335 
336  const Channel & srcAlpha = srcTray.getAlpha();
337  Channel & dstAlpha = dstTray.getAlpha();
338 
340  Point2D<int> p;
341  Point2D<int> pTest;
342  Point2D<int> pWin;
343 
344  // proximity (inverted distance)
345  dist_t dWin;
346  dist_t dTest;
347 
348  // Current position
349  size_t address;
350  // Winning position
351  size_t addressWin;
352 
353  const size_t K = std::min(srcTray.size(), dstTray.size());
354 
355  const Range<int> & xRange = coordinateHandler.getXRange();
356  const Range<int> & yRange = coordinateHandler.getYRange();
357 
358  mout.debug() << "main loop, K=" << K << mout.endl;
359 
360  for (p.y=yRange.max; p.y>=0; --p.y){
361  for (p.x=xRange.max; p.x>=0; --p.x){
362 
363  address = dstAlpha.address(p.x, p.y);
364 
365  dWin = srcAlpha.get<dist_t>(address);
366  pWin.setLocation(p);
367 
368  for (const DistanceElement & elem: chain){
369  pTest.setLocation(p.x + elem.diff.x, p.y + elem.diff.y);
370  coordinateHandler.handle(pTest);
371  dTest = this->distanceModel.decrease(dstAlpha.get<dist_t>(pTest), elem.coeff);
372  if (dTest > dWin){
373  dWin = dTest;
374  pWin = pTest;
375  }
376  }
377 
378 
379  // Finally, update target image (if stronger data found)
380  if (dWin > 0.0){
381 
382  dstAlpha.put(address, dWin);
383 
384  addressWin = dstAlpha.address(pWin.x, pWin.y);
385 
386  if (addressWin != address){
387  for (size_t k=0; k<K; ++k)
388  dstTray.get(k).put(address,dstTray.get(k).get<dist_t>(addressWin));
389  }
390  else {
391  for (size_t k=0; k<K; ++k)
392  dstTray.get(k).put(address,srcTray.get(k).get<dist_t>(address));
393  }
394 
395  }
396 
397  }
