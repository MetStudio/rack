224 template <class T>
225 Range<int> DistanceTransformOp<T>::getHorzRange(const CoordinateHandler2D & coordinateHandler) const {
226 
227  Range<int> xRange = coordinateHandler.getXRange();
228  const Bidirectional<float> & radiusHorz = getDistanceModel().getRadiusHorz();
229 
230  if (coordinateHandler.policy.xUnderFlowPolicy == CoordinatePolicy::WRAP)
231  xRange.min -= radiusHorz.backward;
232 
233  if (coordinateHandler.policy.xOverFlowPolicy == CoordinatePolicy::WRAP)
234  xRange.max += radiusHorz.forward;
235 
236  return xRange;
237 }

284 void DistanceTransformOp<T>::traverseDownRight(const Channel &src, Channel &dst) const
285 {
286 
287  Logger mout(getImgLog(), __FILE__, __FUNCTION__);
288  //mout.debug("start" );
289 
290  //const DistanceModel & distModel = getDistanceModel();
291  mout.debug2("distModel:", this->distanceModel);
292 
293  DistanceNeighbourhood chain;
294  this->distanceModel.createChain(chain, true);
295 
296  mout.warn("coordPolicy: ", src.getCoordinatePolicy());
297 
298  CoordinateHandler2D coordinateHandler(src); // TODO: change param to ImageConf
299  mout.debug2("coordHandler:", coordinateHandler);
300 
301 
302  const ValueScaling src2dst(dst.getConf().getTypeMax<double>()/src.getConf().getTypeMax<double>(), 0);
303  //const ValueScaling src2dst(src.getScaling(), dst.getScaling());
304 
305  for (double d: {0.0, 0.5, 1.0, 255.0, 65535.0}){
306  mout.special(d, " =>\t", src2dst.fwd(d), " inv:", src2dst.inv(d));
307  }
308 
309  // proximity (inverted distance)
310  dist_t d;
311  dist_t dTest;
312 
313  // Point in the source image
314  Point2D<int> pTest;
315 
316  // Point in the target image
317  Point2D<int> p;
318  Point2D<int> pSafe;
319 
320  Range<int> xRange = getHorzRange(coordinateHandler);
321  Range<int> yRange = getVertRange(coordinateHandler);
322  mout.special(xRange, " - ", yRange);
323 
324  // Experimental (element horz/vert topology not yet implemented)
325  // Possibly wrong... not interchangible due to to scanning element geometry?
326  /*
327  bool HORZ = true;
328  const Range<int> & inner = HORZ ? xRange : yRange;
329  const Range<int> & outer = HORZ ? yRange : xRange;
330  int & innerValue = HORZ ? p.x : p.y;
331  int & outerValue = HORZ ? p.y : p.x;
332  mout.debug2("outer range:" , outer );
333  mout.debug2("inner range:" , inner );
334  */
335  //Range<int>
336 
337  // Todo: extended area, needs coordinateHandler.handle(p);?
338  for (p.y=yRange.min; p.y<=yRange.max; ++p.y){
339  for (p.x=xRange.min; p.x<=xRange.max; ++p.x){
340 
341  pSafe.setLocation(p);
342  if (coordinateHandler.handle(pSafe) == CoordinateHandler2D::IRREVERSIBLE)
343  continue;
344 
345  // Take (converted) source value as default
346  d = src2dst.fwd(src.get<dist_t>(pSafe));
347 
348  // Compare to previous value
349  dTest = dst.get<dist_t>(pSafe);
350  if (dTest > d){
351  d = dTest;
352  }
353 
354  for (const DistanceElement & elem: chain){
355  pTest.setLocation(pSafe.x + elem.diff.x, pSafe.y + elem.diff.y);
356  coordinateHandler.handle(pTest);
357  dTest = this->distanceModel.decrease(dst.get<dist_t>(pTest), elem.coeff);
358  if (dTest > d){
359  d = dTest;
360  }
361  }
362 
363  if (d > 0){
364  dst.put(pSafe,d);
365  }
366 
367  };
368  };
