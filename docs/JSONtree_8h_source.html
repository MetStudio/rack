136 void JSONtree::readTree(T & t, std::istream & istr){
137 
138  drain::Logger log("JSON", __FUNCTION__);
139 
140  if (!istr){
141  log.error() << "File read error" << log.endl;
142  return;
143  }
144 
145  char c;
146 
147  TextReader::skipChars(istr, " \t\n\r");
148  c = istr.get();
149  if (c != '{'){
150  //std::cerr << "Fail: " << c << '\n';
151  log.error() << "Syntax error: read '" << c << "' when expecting '{'" << log.endl;
152  return;
153  }
154 
155  std::string key;
156  std::string value;
157  bool completed = false;
158 
159  while (istr){
160 
161  TextReader::skipChars(istr, " \t\n\r");
162 
163  c = istr.get();
164 
165  if (c == '"'){ // New entry
166 
167  key = TextReader::scanSegment(istr, "\"");
168  // log.warn() << " then1: " << (char)istr.peek() << log.endl;
169  //istr.get(); // swallow terminator
170  // log.warn() << " then2: " << (char)istr.peek() << log.endl;
171 
172  typename T::node_t & vmap = t.data;
173 
174  TextReader::skipWhiteSpace(istr); // Chars(istr, " \t\n\r");
175  // log.warn() << " then3: " << (char)istr.peek() << log.endl;
176  c = istr.get();
177  // log.warn() << " then4: " << (char)istr.peek() << log.endl;
178 
179  if (c == ':'){
180  TextReader::skipWhiteSpace(istr) ; // Chars(istr, " \t\n\r");
181  c = istr.peek();
182 
183  if (c == '{'){
184  log.debug3() << "Reading object '" << key << "'" << log.endl;
186  JSONtree::readTree(t[key], istr);
187  }
188  else {
189  log.debug3() << "Reading value '" << key << "'" << log.endl;
190  JSONreader::readValue(istr, vmap[key]);
191  //ValueReader::scanValue(istr, vmap[key]);
192  }
193  completed = true;
194  }
195  else {
196  log.error() << "Syntax error: read \"" << key << "\" followed by '" << c << "' when expecting object {...}, string \"...\", array [...], or number" << log.endl;
197  return;
198  }
199  }
200  else if (c == '}')
201  return;
202  else if (c == ','){
203  if (!completed) // comma encountered after empty segment
204  log.warn() << "empty section after key=" << key << log.endl;
205  completed = false; // trap for subsequent check
206  }
207  else { // TODO: warn if comma encountered after empty
208  log.error() << "Syntax error: char '" << c << "', expected '\"', '}' or ','" << log.endl;
209  }
210 
211  }
212 
213 }
