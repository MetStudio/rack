139  template <class T>
140  std::ostream & toStream(std::ostream & ostr, const std::map<std::string,T> &m, bool clear = false) const {
141  ostr.fill(fill);
142  for (StringMapper::const_iterator it = begin(); it != end(); it++){
143  if (it->isVariable()){
144  typename std::map<std::string, T >::const_iterator mit = m.find(*it);
145  if (mit != m.end()){
146  ostr.width(width);
147  ostr << mit->second;
148  }
149  else if (!clear)
150  ostr << variablePrefix << *it << variableSuffix;
151  }
152  else
153  ostr << *it;
154  };
155  return ostr;
