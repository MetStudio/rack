421 bool DataSelector::getPathsNEW(const HI5TREE &src, T & pathContainer, const drain::RegExp & quantityRE, ODIMPathElem::group_t groupFilter, const ODIMPath & path) const {
422 
423  drain::Logger mout(getName(), __FUNCTION__);
424 
425  mout.debug(3) << "considering '" << path << "/' -> ..." << mout.endl;
426 
427  // ODIM struct is needed to communicate keys for maps (elangle or quantity) to addPathT
428  PolarODIM odim;
429 
430  if (path.empty())
431  mout.debug(3) << "empty path (ok)" << mout.endl;
432  else if (!src.hasPath(path)){
433  mout.warn() << "data structure has no path:" << path << mout.endl;
434  return false;
435  }
436 
437  // Current search point
438  const HI5TREE & s = src(path); // =src, if path empty
439 
440 
441  // Selector contains a rule for desired quantity. (This also indicates that quantityRE is set).
442  const bool QUANTITY_CONSTRAINT = !quantity.empty();
443 
444 
445  // Changed to true, if a descendant contains the desired quantity.
446  bool quantityGroupOK = QUANTITY_CONSTRAINT ? false : true;
447 
448  //const bool IS_ROOT = (path.size() == 1);
449  std::set<ODIMPathElem> datasets;
450 
451  for (HI5TREE::const_iterator it = s.begin(); it != s.end(); ++it) {
452 
453  //ODIMPathElem child(it->first);
454  const ODIMPathElem & currentElem = it->first;
455  //mout.debug(3) << "*it='" << it->first << "' (" << it->first.group << "),\t currentElem=" << "'" << currentElem <<"' (" << currentElem.group << ") include=" << currentElem.belongsTo(groupFilter) << mout.endl;
456  mout.debug(3) << "currentElem=" << "'" << currentElem <<"' (" << currentElem.group << ") incl=" << currentElem.belongsTo(groupFilter) << mout.endl;
457 
458 
459  if (currentElem.is(ODIMPathElem::DATASET)){ // what about quality?
460  if (!dataset.isInside(currentElem.index)){
461  mout.debug(3) << "dataset " << currentElem.index << " not in [" << dataset << "], skipping" << mout.endl;
462  continue;
463  }
464  }
465  else if (currentElem.is(ODIMPathElem::DATA)){ // what about quality?
466  if (!data.isInside(currentElem.index)){
467  mout.debug(3) << "data " << currentElem.index << " not in [" << data << "], skipping" << mout.endl;
468  continue;
469  }
470  }
471 
472  const hi5::NodeHi5 & node = it->second.data;
473 
474  // Currently, there is no switch to exclude/include no-save (temporary) structures.
475  if (node.noSave){
476  mout.debug(3) << "noSave data, ok: " << it->first << mout.endl;
477  //continue;
478  }
479 
480 
481  // In indexed DATASET/QUALITY/DATA groups, images are the metadata containers
482  const drain::image::Image & d = node.dataSet;
483  //mout.note() << "reconsidering " << currentElem << ':' << d.properties["what:quantity"] << mout.endl;
484 
485  if (currentElem.is(ODIMPathElem::DATASET)){ // what about quality?
486  if (d.properties.hasKey("where:elangle")){
487  if (!elangle.isInside(d.properties["where:elangle"])){
488  mout.debug() << "outside elangle range"<< mout.endl;
489  continue;
490  }
491  }
492  }
493 
494  bool quantityOK = true;
495  if (QUANTITY_CONSTRAINT){
496 
497  // IS this check needed (here)? Consider above: QUANTITY_CONSTRAINT && (currentElem.is(ODIMPathElem::DATA) || currentElem.is(ODIMPathElem::QUALITY) )
498  if (currentElem.is(ODIMPathElem::ARRAY)){
499  if (groupFilter & ODIMPathElem::ARRAY){
500  mout.warn() << "path " << path << '/' << currentElem << " will not be detected; ARRAY does not support quantity (request=" << quantity << ")" << mout.endl;
501  }
502  }
503 
504  if (currentElem.is(ODIMPathElem::DATA) || currentElem.is(ODIMPathElem::QUALITY) ){
505 
506  if (d.properties.hasKey("what:quantity")){
507 
508  quantityOK = quantityRE.test(d.properties["what:quantity"]);
509  if (quantityOK){
510  quantityGroupOK = true;
511  mout.debug() << it->first << ":\t found quantity '" << d.properties["what:quantity"] << "'" << mout.endl;
512  }
513  else {
514  mout.debug(1) << it->first << ":\t quantity '" << quantity << "' !~ '" << d.properties["what:quantity"] << "'" << mout.endl;
515  // allow sub-QIND, so do not: continue;
516  }
517  }
518  else {
519  if (currentElem.belongsTo(groupFilter))
520  mout.warn() << "testing '" << quantity << "': no what:quantity in " << path << '/' << it->first << mout.endl;
521  // else check currentElem.is(ODIMPathElem::QUALITY) and QIND?
522  }
523  }
524  }
525 
526 
527 
528  ODIMPath p(path);
529  p << currentElem;
530 
532  //mout.warn() << "descending to " << p << mout.endl;
533  T descendantPaths;
534  const bool quantitySubtreeOK = getPathsNEW(src, descendantPaths, quantityRE, groupFilter, p) ; // note: original "root" src
535 
536  if (quantitySubtreeOK)
537  quantityGroupOK = true;
538 
539 
541  bool accept = true;
542 
543  if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::QUALITY)){
544  if (!quantityOK)
545  accept = false;
546  }
547  else if (currentElem.is(ODIMPathElem::DATASET)){
548  if (!quantitySubtreeOK)
549  accept = false;
550  }
551 
552  if (accept){
553 
554  // COUNTER. Otherways accepted now, only check the counter. Counter must not be incremented before this point.
555  if (currentElem.is(ODIMPathElem::DATASET)){
556  datasets.insert(currentElem);
557  mout.debug(2) << "now dataset count=" << datasets.size() << " latest=" << currentElem << mout.endl;
558  if (datasets.size() > count){
559  mout.debug(1) << " count=" << count << " exceeded, returning" << mout.endl;
560  return true; // = dump this currentElem and its descendants
561  }
562  }
563 
564  if (currentElem.belongsTo(groupFilter)){
565  mout.debug(1) << "accepted " << p << mout.endl;
566  odim.clear();
567  odim.copyFrom(d); // OK, uses true type ie. full precision, also handles img type
568  addPathT(pathContainer, odim, p);
569  mout.debug(2) << "pathContainer size=" << pathContainer.size() << mout.endl;
570  }
571  }
572 
573 
574  // Note: all the descendantPaths (if exist) are confirmed already in the inner call.
575  mout.debug(3) << "adding descendants: " << descendantPaths.size() << mout.endl;
576 
577  // Append descendant paths now (all have been accepted already)
578  for (typename T::const_iterator it = descendantPaths.begin(); it != descendantPaths.end(); ++it){
579  pathContainer.insert(pathContainer.end(), *it);
580  //pathContainer.insert(*it);
581  }
582 
583  //mout.warn() << "pathContainer size=" << pathContainer.size() << mout.endl;
584 
585  }
586 
587  return quantityGroupOK;
588 }
