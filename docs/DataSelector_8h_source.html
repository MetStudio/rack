471 bool DataSelector::getPaths3(const Hi5Tree & src, T & pathContainer, const ODIMPath & path) const {
472 
473  drain::Logger mout(__FUNCTION__, __FILE__);
474 
475  if (path.empty())
476  mout.debug(1) << "matcher: " << pathMatcher << mout.endl;
477 
478  // Current search point
479  const Hi5Tree & s = src(path); // =src, if path empty
480 
481  // ODIM struct is needed to communicate keys for maps (elangle or quantity) to addPathT
482  PolarODIM odim;
483 
485  std::set<ODIMPathElem::index_t> dataSetIndices;
486 
487  const bool quantityRequired = quantityRegExp.isSet() || qualityRegExp.isSet();
488  //bool quantityChecked = false;
489  bool quantityFound = false;
490  //bool resultOK = true;
491 
492  for (Hi5Tree::const_iterator it = s.begin(); it != s.end(); ++it) {
493 
494  bool quantityOk = !quantityRequired; //
495 
496  const ODIMPathElem & currentElem = it->first;
497  mout.debug() << "currentElem='" << currentElem << "'" << mout.endl;
498 
499  const drain::image::Image & d = it->second.data.dataSet; // for ODIM
500  const drain::FlexVariableMap & v = d.getProperties();
501 
502  // Check ELANGLE (in datasets) or quantity (in data/quality)
503  if (currentElem.is(ODIMPathElem::DATASET)){ // what about quality?
504 
505  if (v.hasKey("where:elangle")){
506  if (!elangle.contains(v["where:elangle"])){
507  mout.debug() << "outside elangle range" << mout.endl;
508  continue;
509  }
510  }
511 
512  if (dataSetIndices.size() >= count){
513  mout.debug() << "count (" << dataSetIndices.size() << ">=" << count << ") already completed for " << currentElem << mout.endl;
514  continue;
515  }
516 
517  }
518  else if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::QUALITY)){
519 
520  if (quantityRequired){
521 
522  if (v.hasKey("what:quantity")){
523 
524  const std::string quantity = v["what:quantity"].toStr();
525 
526  if (qualityRegExp.isSet()){ // NOTE: at least, do not test quantity after this!
527  //quantityOk = false;
528  if (currentElem.is(ODIMPathElem::QUALITY) && qualityRegExp.test(quantity)){
529  mout.debug(2) << "matching QUALITY quantity [" << quantity << "], skipping" << mout.endl;
530  quantityOk = true;
531  quantityFound = true;
532  }
533  }
534  else if (quantityRegExp.test(quantity)){
535  mout.debug(1) << "OK quantity matches: " << quantity << ": " << path << '|' << currentElem << mout.endl;
536  quantityOk = true;
537  quantityFound = true;
538  }
539  else {
540  mout.debug(2) << "unmatching DATA quantity [" << quantity << "], skipping" << mout.endl;
541  //NO continue; have to descend for quality (QIND) below!
542  //quantityOk = false;
543  }
544 
545  }
546  else {
547  if (currentElem.is(ODIMPathElem::DATA))
548  mout.warn();
549  else
550  mout.info();
551  mout << "quantity missing in (image) metadata of " << path << '/' << currentElem << mout.endl;
552  }
553  }
554  }
555  else {
556  //mout.warn() << "NOT testing: " << currentElem << mout.endl;
557  }
558 
559 
560  ODIMPath p(path);
561  p << currentElem;
562 
563 
564  if (currentElem.is(ODIMPathElem::DATASET)){
565 
566  if (pathMatcher.match(p) && (dataSetIndices.size() < count)){
567 
568  typename T::iterator tit = pathContainer.end();
569  addPath3(pathContainer, v, p); // accept tentatively, maybe cancelled below
570 
571  // RECURSION
572  quantityOk = getPaths3(src, pathContainer, p);
573  if (quantityOk || !quantityRequired){
574  mout.debug() << "yes, was ok: " << p << mout.endl;
575  }
576  else {
577  mout.debug() << "oops, removing: " << p << mout.endl;
578  pathContainer.erase(--tit);
579  }
580  // check over flow
581  }
582  else {
583  // not collecting DATASET's but traversing
584  getPaths3(src, pathContainer, p);
585  }
586 
587  }
588  else if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::QUALITY)){
589  //mout.warn() << "descending D/Q :" << currentElem << mout.endl;
590  if (quantityOk && pathMatcher.match(p)){
591  addPath3(pathContainer, v, p);
592  }
593  getPaths3(src, pathContainer, p);
594  }
595  else {
597  if (pathMatcher.match(p)){
598  addPath3(pathContainer, v, p);
599  }
600  getPaths3(src, pathContainer, p); // note: original "root" src
601  quantityOk = true; //?
602  }
603 
604  if (quantityOk){
605  const ODIMPathElem & e = p.front();
606  if (pathMatcher.matchElem(e, true)){
607  mout.debug(1) << "stem: " << e << mout.endl;
608  dataSetIndices.insert(e.getIndex());
609  }
610  //mout.warn() << "datasets: " << drain::StringTools::join(dataSetIndices,'*') << '=' <<dataSetIndices.size() << ':' << p.front().getIndex() << '|' << p << mout.endl;
611  }
612 
613  }
614  //mout.warn() << "quantityFound " << p << " -> ..."<< mout.endl;
615 
616  return (quantityFound || !quantityRequired);
617 }
