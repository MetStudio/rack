310 
311  const typename T::container_t & children = tree.getChildren();
312 
313  // Indent

322  else {
323  ostr << '<' << tree->getTag() << ' ';
324  // TODO check GDAL XML
325  if (!defaultTag.empty())
326  attribToStream(ostr, "name", defaultTag);

328 
329  // if (tree->id >= 0){

331  ostr << "class=\"";

335  // attribToStream(ostr, "id", tree->id); // problem for palette?
336  //ostr << "id=\"" << tree.data.id << '"' << ' ';
337  // }
338 
340  for (const typename T::node_data_t::key_t & key: tree.data.getKeyList()){
341  std::stringstream sstr;
342  sstr << tree.data[key]; // consider checking 0, not only empty string "".
343  if (!sstr.str().empty()){
344  attribToStream(ostr, key, sstr.str());
345  // ostr << key << "=\"" << sstr.str() << '"' << ' ';
346  }
347 
348  }
349 
350  if (tree->isComment()){
351  ostr << " /-->\n";
352  }
353  else if ((children.size() == 0) && tree->ctext.empty() ){ // OR no ctext!
354  // close TAG
