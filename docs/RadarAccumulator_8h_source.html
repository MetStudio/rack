162 void RadarAccumulator<AC,OD>::addData(const pdata_src_t & srcData, const pdata_src_t & srcQuality, double weight, int i0, int j0){
163 
164  drain::Logger mout("RadarAccumulator", __FUNCTION__);
165 
166  if (!srcQuality.data.isEmpty()){
167  mout.info() << "Quality data available with input; using quality as weights in compositing." << mout.endl;
168  DataCoder converter(srcData.odim, srcQuality.odim);
169  // uses also DataCoder::undetectQualityCoeff
170  AC::addData(srcData.data, srcQuality.data, converter, weight, i0, j0);
171  }
172  else {
173  mout.info() << "No quality data available with input, ok." << mout.endl;
174  ODIM qualityOdim;
175  getQuantityMap().setQuantityDefaults(qualityOdim, "QIND");
176 
177  DataCoder converter(srcData.odim, qualityOdim);
178  // uses also DataCoder::undetectQualityCoeff
179  AC::addData(srcData.data, converter, weight, i0, j0);
180  }
181 
182  counter += std::max(1L, srcData.odim.ACCnum);
183  odim.updateLenient(srcData.odim); // Time, date, new
184 
185  //mout.note() << "after: " << this->odim << mout.endl;
186 
187 }

201 template <class AC, class OD>
202 bool RadarAccumulator<AC,OD>::checkCompositingMethod(const ODIM & dataODIM) const {
203 
204  drain::Logger mout("RadarAccumulator", __FUNCTION__);
205 
206  mout.debug() << "start, quantity=" << dataODIM.quantity << mout.endl;
207 
208  const AccumulationMethod & rule = this->getMethod();
209  if (rule.name == "WAVG"){
210 
211  const drain::ReferenceMap & wavgParams = rule.getParameters();
212  const double p = wavgParams.get("p", 1.0);
213  const double p2 = p/2.0;
214  mout.info() << rule << ", min="<< dataODIM.getMin() << mout.endl;
215 
216  if ((p2 - floor(p2)) == 0.0){
217  mout.info() << "WAVG with p=" << p << " = 2N, positive pow() results" << mout.endl;
218  const Quantity & q = getQuantityMap().get(dataODIM.quantity);
219  if (q.hasUndetectValue()){
220  const double bias = wavgParams.get("bias", 0.0);
221  if (bias > q.undetectValue){
222  mout.warn() << "WAVG with p=" << p <<" = 2N, undetectValue=" << q.undetectValue << " < bias="<< bias << mout.endl;
223  mout.warn() << "consider adjusting bias("<< bias <<") down to quantity (" << dataODIM.quantity << ") zero: " << q.undetectValue << mout.endl;
224  //rule.setParameter("bias", q.undetectValue);
225  return true;
226  }
227  }
228  }
229  }
230 
231  return false;

304  //odimQuality.undetect = 256;

311  //const QuantityMap & qm = getQuantityMap();

314  mout.note() << "found quantyConf[" << odimQuality.quantity << "], OK" << mout.endl;
315  mout.warn() << EncodingODIM(odimQuality) << mout.endl;
316  }
317  else {
318  odimQuality.scaling.scale *= 20.0; // ?
319  //const std::type_info & t = Type::getType(odimFinal.type);
320  odimQuality.scaling.offset = round(drain::Type::call<drain::typeMin, double>(t) + drain::Type::call<drain::typeMax, double>(t))/2.0;
321  //odimQuality.offset = round(drain::Type::call<drain::typeMax,double>(t) + drain::Type::getMin<double>(t))/2.0; // same as data!
322  mout.warn() << "quantyConf[" << odimQuality.quantity << "] not found, using somewhat arbitary scaling:" << mout.endl;
323  mout.warn() << EncodingODIM(odimQuality) << mout.endl;
324  }
325  break;
326  default:
327  mout.error() << "Unsupported field code: '" << field << "'" << mout.endl;
328  break;
329  }
330 
331  mout.debug() << "extracting field " << field << mout.endl;
332 
333 
334  if (type == DATA){
335  dataPath << "data" << ++dCounter;
336  mout.debug() << "target: " << EncodingODIM(odimFinal) << mout.endl;

341  }
342 
343

348

351  pdata_dst_t & dstData = (type == DATA) ? dstProduct.getData(odimFinal.quantity) : dstProduct.getQualityData(odimQuality.quantity);
352  mout.debug(3) << "dstData: " << dstData << mout.endl;
353 
354  //DataDst dstData(dataGroup); // FIXME "qualityN" instead of dataN creates: /dataset1/qualityN/quality1/data
355  //mout.warn() << "odimFinal: " << odimFinal << mout.endl;
356

358  mout.debug() << "DATA/" << field << mout.endl;
359  //mout.warn() << dstData.odim << mout.endl;
360  dstData.odim.importMap(odimFinal);
361  dstData.data.setType(odimFinal.type);
362  //mout.debug() << "quantity=" << dstData.odim.quantity << mout.endl;
