134 
135  drain::Logger mout("RadarAccumulator", __FUNCTION__);
136 
137  if (!srcQuality.data.isEmpty()){
138  mout.info() << "Quality data available with input; using quality as weights in compositing." << mout.endl;
139  DataCoder converter(srcData.odim, srcQuality.odim);
140  // uses also DataCoder::undetectQualityCoeff
141  AC::addData(srcData.data, srcQuality.data, converter, weight, i0, j0);

145  ODIM qualityOdim;

149  // uses also DataCoder::undetectQualityCoeff
150  AC::addData(srcData.data, converter, weight, i0, j0);
151  }
152 
153  counter += std::max(1L, srcData.odim.ACCnum);
154  odim.update(srcData.odim); // Time, date, new

173 bool RadarAccumulator<AC,OD>::checkCompositingMethod(const ODIM & dataODIM) const {
174 
175  drain::Logger mout("RadarAccumulator", __FUNCTION__);
176 
177  mout.debug() << "start, quantity=" << dataODIM.quantity << mout.endl;
178 
179  const AccumulationMethod & rule = this->getMethod();
180  if (rule.name == "WAVG"){
181 
182  const drain::ReferenceMap & wavgParams = rule.getParameters();
183  const double p = wavgParams.get("p", 1.0);
184  const double p2 = p/2.0;
185  mout.info() << rule << ", min="<< dataODIM.getMin() << mout.endl;
186 
187  if ((p2 - floor(p2)) == 0.0){
188  mout.info() << "WAVG with p=" << p << " = 2N, positive pow() results" << mout.endl;
189  const Quantity & q = getQuantityMap().get(dataODIM.quantity);
190  if (q.hasUndetectValue){
191  const double bias = wavgParams.get("bias", 0.0);
192  if (bias > q.undetectValue){
193  mout.warn() << "WAVG with p=" << p <<" = 2N, undetectValue=" << q.undetectValue << " < bias="<< bias << mout.endl;
194  mout.warn() << "consider adjusting bias("<< bias <<") down to quantity (" << dataODIM.quantity << ") zero: " << q.undetectValue << mout.endl;
195  //rule.setParameter("bias", q.undetectValue);
196  return true;
197  }
198  }
199  }
200  }
201 
202  return false;

211 
212  drain::Logger mout("RadarAccumulator", __FUNCTION__);
213  // mout.warn() << "is root?\n" << dst << mout.endl;

221

227

229

232 
233  std::stringstream dataPath;
234 
235  datatype type = DATA;
236  char field = quantities.at(i);
237  switch (field) {
238  case 'm': // ???
239  case 'D':
240  case 'p': // ???
241  mout.warn() << "non-standard layer code; use 'd' for 'data' instead" << mout.endl;
242  // no break
243  case 'd':
244  type = DATA;
245  odimFinal = odimOut; // consider update
246  qm.setQuantityDefaults(odimQuality, "QIND");
247  odimQuality.quantity = "QIND";
248  break;
249  case 'c':
250  type = QUALITY;
251  qm.setQuantityDefaults(odimQuality, "COUNT");
252  odimQuality.quantity = "COUNT";
253  break;
254  case 'C':
255  type = QUALITY;
256  qm.setQuantityDefaults(odimQuality, "COUNT", "d");
257  odimQuality.quantity = "COUNT";
258  break;
259  case 'w':
260  // no break
261  type = QUALITY;
262  odimFinal = odimOut; // (Because converted needs both data and weight to encode weight?)
263  qm.setQuantityDefaults(odimQuality, "QIND", "C");
264  odimQuality.quantity = "QIND";
265  //odimQuality.undetect = 256;
266  //odimQuality.nodata = -1; // this is good, because otherwise nearly-undetectValue-quality CAPPI areas become no-data.
267  break;
268  case 'W':
269  mout.warn() << "experimental: quality [QIND] type copied from data [" << odimOut.quantity << ']' << mout.endl;
270  // no break
271  type = QUALITY;
272  odimFinal = odimOut; // (Because converted needs both data and weight to encode weight?)
273  qm.setQuantityDefaults(odimQuality, "QIND", odimOut.type);
274  odimQuality.quantity = "QIND";
275  //odimQuality.undetect = 256;
276  //odimQuality.nodata = -1; // this is good, because otherwise nearly-undetectValue-quality CAPPI areas become no-data.
277  break;
278  case 's':
279  type = QUALITY;
280  odimQuality = odimOut;
281  odimQuality.quantity += "DEV";
282  odimQuality.gain *= 20.0; // ?
283  //const std::type_info & t = Type::getType(odimFinal.type);
284  odimQuality.offset = round(drain::Type::call<drain::typeMin, double>(t) + drain::Type::call<drain::typeMax, double>(t))/2.0;
285  //odimQuality.offset = round(drain::Type::call<drain::typeMax,double>(t) + drain::Type::getMin<double>(t))/2.0; // same as data!
286  break;
287  default:
288  mout.error() << "Unsupported field code: '" << field << "'" << mout.endl;
289  break;
290  }
291 
292  mout.info() << "extracting field " << field << mout.endl;
293 
294 
295  if (type == DATA){
296  dataPath << "data" << ++dCounter;
297  mout.debug() << "target: " << EncodingODIM(odimFinal) << mout.endl;
298  }
299  else if (type == QUALITY){
300  dataPath << "quality" << ++qCounter;
301  mout.debug() << "target: " << EncodingODIM(odimQuality) << mout.endl;
302  }
303 
304 
305  if (odimFinal.quantity.empty()){
306  odimFinal.quantity = "UNKNOWN"; // for example --cPlotFile carries no information on quantity
307  mout.note() << "quantity=" << odimFinal.quantity << mout.endl;
308  }
309 
310  //PlainData<DstType<OD> >
311  mout.debug() << "searching dstData... DATA=" << (type == DATA) << mout.endl;
312  pdata_dst_t & dstData = (type == DATA) ? dstProduct.getData(odimFinal.quantity) : dstProduct.getQualityData(odimQuality.quantity);
313  mout.debug(3) << "dstData: " << dstData << mout.endl;
314 
315  //DataDst dstData(dataGroup); // FIXME "qualityN" instead of dataN creates: /dataset1/qualityN/quality1/data
316  //mout.warn() << "odimFinal: " << odimFinal << mout.endl;
317 
318  if (type == DATA){
319  mout.debug() << "DATA/" << field << mout.endl;
320  //mout.warn() << dstData.odim << mout.endl;
321  dstData.odim.importMap(odimFinal);
322  dstData.data.setType(odimFinal.type);
323  //mout.debug() << "quantity=" << dstData.odim.quantity << mout.endl;
324  }
325  else {
326  mout.debug() << "QUALITY/" << field << mout.endl;
327  dstData.odim.updateFromMap(odimQuality);
328  //dstData.odim.importMap(odimQuality);
329  }
330 
331  DataCoder converter(odimFinal, odimQuality); // (will use only either odim!)
332 
333  mout.debug() << "converter: " << converter.toStr() << mout.endl;
334 
335  this->Accumulator::extractField(field, converter, dstData.data);
336 
337  //mout.debug() << "dstData: " << dstData.odim << mout.endl;
338  mout.debug() << "updating local tree attributes" << mout.endl;
339  //hi5::Writer::writeFile("test0.h5", dstProduct.tree);
340  //@= dstData.updateTree();
341  //std::cerr << __FUNCTION__ << ':' << dstData.tree << std::endl;
342  //hi5::Writer::writeFile("test1.h5", dstProduct.tree);
343 
344  }
345 
346 
347  mout.debug() << "updating local tree attributes" << mout.endl;
348 
349 
350  //mout.debug() << "Finished " << accumulator.getMethod().name << mout.endl;
351  mout.debug() << "finished" << mout.endl;
352 
353 }
