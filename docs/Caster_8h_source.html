567 void Caster::put<std::string>(void *p, const std::string & x) const {
568  //put(p, x.c_str()); Dangerous, if F = std::string
569 
570  //std::cerr << "Laita std::string x=" << x << std::endl;
571  if (getType() == typeid(std::string))

573  else if (getType() == typeid(char)){

581  }
582  }
583  else
584  put(p, x.c_str());

655 template <> inline
656 std::string Caster::get<std::string>(const void *p) const {
657 
658  // void is handled by toOStr, but this is faster.
659  if (getType() == typeid(void)){
660  static const std::string empty;
661  return empty;
662  }
663  else if (getType() == typeid(char)){
664  // std::cout << "note: experimental char to str\n";
665  return std::string(1, *(char *)p); // NOTE: this does not handle char array of more elements!
666  }
667  else if (getType() == typeid(bool)){
668  return *(const bool *)p ? "true" : "false";
669  }
670  else if (getType() == typeid(std::string)){
671  return *(const std::string *)p;
672  }
673  else {
674  std::stringstream sstr;
675  toOStream(sstr, p);
676  // sstr << (this->getDouble)(p);
677  return sstr.str();
678  }
679 }
