480 template <class C>
481 const std::type_info & Type::guessArrayType(const C & l){
482 
483 
484  typedef std::set<const std::type_info *> typeset;
485 
486  typeset s;
487  for (typename C::const_iterator it = l.begin(); it != l.end(); ++it) {
488  s.insert(& guessType(*it));
489  }
490 
492  if (s.find(& typeid(std::string)) != s.end())
493  return typeid(std::string);
494 
496  if (s.find(& typeid(double)) != s.end())
497  return typeid(double);
498 
499  if (s.find(& typeid(int)) != s.end())
500  return typeid(int);
501 
503  if (s.find(& typeid(bool)) != s.end())
504  return typeid(bool);
505 
506  // General fallback solution
507  return typeid(std::string);
508 
509 }
