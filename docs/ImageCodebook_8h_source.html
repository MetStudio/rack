86  lookup_t & createLookUp(const std::type_info & type, const ValueScaling & scaling) const { // todo N?
87 
88  drain::Logger mout(__FUNCTION__, __FILE__);
89 
90  //mout.warn() << "first entry: " << sprinter(*pal.begin()) << mout;
91  //mout.warn() << "last entry: " << sprinter(*pal.rbegin()) << mout;
92 
93  lookUp.byteSize = drain::Type::call<drain::sizeGetter>(type);
94 
95  // Todo: add support for signed short and signed char ?
96  if (type == typeid(unsigned short)){ // 1024 entries (not 65536)
97  if ((scaling.scale == 1.0) && (scaling.offset == 0.0)){
98  double first = this->begin()->first;
99  double last = this->rbegin()->first;
100  if ((first >= 0.0) && (last <= 255.0)){
101  lookUp.bitShift = 0;
102  mout.note() << "short int, but not using 16 bits, coping with 256 entries" << mout;
103  }
104  else
105  lookUp.bitShift = 6;
106  }
107  else
108 
109  lookUp.bitShift = 6; // note => 16 - 6 = 10 bits => 1024 entries
110  }
111  else if (type == typeid(unsigned char)){
112  lookUp.bitShift = 0;
113  }
114  else {
115  mout.note() << "not supported for type " << drain::Type::getTypeChar(type) << mout.endl;
116  lookUp.byteSize = 0;
117  lookUp.bitShift = 0;
118  lookUp.clear();
119  return lookUp;
120  }
121 
122  const int n = (1 << (lookUp.byteSize*8 - lookUp.bitShift));
123  mout.debug() << "type=" << drain::Type::getTypeChar(type) << ", creating " << n << " lookup entries" << mout.endl;
124 
125  typename cont_t::const_iterator itLower = this->begin();
126 
127  lookUp.resize(n, itLower);
128 
129  // Signed, because scaling physical values may cause underflow
130  int index, indexLower=0;
131 
133  for (typename cont_t::const_iterator it=this->begin(); it!=this->end(); ++it){
134 
135  index = static_cast<int>(scaling.inv(it->first));
136 
137  if (index < 0){
138  mout.warn() << "underflow threshold " << it->first << " mapped to negative index " << index << " (before bitShift), skipping " << mout.endl;
139  continue;
140  }
141 
142  index = (index >> lookUp.bitShift);
143 
144  if (index >= n){
145  mout.warn() << "overflow: threshold " << it->first << " mapped to index (" << index << ") > max (" << (n-1) << "), skipping " << mout.endl;
146  continue;
147  }
148 
149  if (indexLower < index){
150  mout.debug() << "adding index range [" << indexLower << '-' << index << "[ -> ";
151  mout << "(" << itLower->first << ") => {" << itLower->second << '}';
152  mout << mout.endl;
153  }
154  else {
155  mout.note() << "accuracy loss: skipped entry [" << index << "] => " << it->first << mout.endl;
156  }
157 
159  for (int i=indexLower; i<index; ++i){
160  lookUp[i] = itLower;
161  }
162 
163  indexLower = index;
164  itLower = it;
165 
166  }
167 
168  index = n;
169  if (indexLower < index){
170  mout.debug() << "padding [" << indexLower << '-' << index << "[ -> \t";
171  mout << '[' << itLower->first << "] // " << itLower->second << mout.endl;
172  }
173  for (int i=indexLower; i<index; ++i){
174  lookUp[i] = itLower;
175  }
176 
177  return lookUp;
178  }
