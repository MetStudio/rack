191 template <class T>
192 void StringTools::split(const std::string & str, T & sequence, const std::string &separators, const std::string & trimChars){
193 
194  sequence.clear();
195 
196  const bool TRIM = !trimChars.empty();
197  const std::string::size_type n = str.size();
198 
199  std::string::size_type pos1 = 0; // Start of segment
200  std::string::size_type pos2 = std::string::npos; // End of segment (index of last char + 1)
201 
203  if (TRIM){
204  // std::cerr << __FUNCTION__ << " initial trim: " << str << '\n';
205  StringTools::trim(str, pos1, pos2, trimChars);
206  }
207 
208  // Tmp
209  typename T::value_type tmpValue;
210 
211  if (separators.empty()){ // = no split! :-)
212 
213  if (TRIM){
214  sequence.insert(sequence.end(), StringTools::convert(str.substr(pos1, pos2-pos1), tmpValue));
215  }
216  else
217  sequence.insert(sequence.end(), StringTools::convert(str, tmpValue));
218 
219  //sequence.push_back(tmpValue); not in set
220  return;
221  }
222  else {
223 
224  // Index of the next separator position
225  std::string::size_type pos = pos1; // "irreversible"
226 
227  do {
228 
229  pos1 = pos;
230  pos = str.find_first_of(separators, pos);
231  if (pos == std::string::npos){
232  pos2 = n;
233  pos = n; // last
234  }
235  else {
236  pos2 = pos;
237  ++pos; // for the next round
238  }
239 
240  //std::cerr << __FUNCTION__ << ":: SEG = [" << pos1 << ',' << pos2 << '['<< "'...\n";
241 
242  if (TRIM){
243  // std::cerr << __FUNCTION__ << ": trimming: '" << str.substr(pos1, pos2-pos1) << "' to ...\n"; // << "' [" << pos1 << ',' << pos2 << '['
244  if (StringTools::trim(str, pos1, pos2, trimChars)){
245  //std::cerr << __FUNCTION__ << " element trim: => => [" << pos1 << ',' << pos2 << '[' << " = '" << str.substr(pos1, pos2-pos1) << "'\n";
246  // std::cerr << __FUNCTION__ << ": ... to [" << pos1 << ',' << pos2 << '[' << " = '" << str.substr(pos1, pos2-pos1) << "'\n";
247  sequence.insert(sequence.end(), StringTools::convert(str.substr(pos1, pos2-pos1), tmpValue));
248  }
249  else {
250  sequence.insert(sequence.end(), StringTools::convert("", tmpValue));
251  }
252  }
253  else {
254  //std::cerr << __FUNCTION__ << " element seg = '" << str.substr(pos1, pos2-pos1) << "'\n";
255  sequence.insert(sequence.end(), StringTools::convert(str.substr(pos1, pos2-pos1), tmpValue));
256  //sequence.insert(sequence.end(), StringTools::convert(str.substr(pos1, pos2-pos1), tmpValue));
257  }
258 
259  //pos = pos + 1;
260 
261  } while (pos != n);
262  }
263 }
