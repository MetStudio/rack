246 template <class T, class C>
247 void StringTools::split(const std::string & str, T & sequence, const C & separators, const std::string & trimChars){
248 
249  sequence.clear();
250 
251  const bool TRIM = !trimChars.empty();
252  const std::string::size_type n = str.size();
253 
254  std::string::size_type pos1 = 0; // Start of segment
255  std::string::size_type pos2 = n; // std::string::npos; // End of segment (index of last char + 1)
256 
257  if (false){ //separators.empty()){ // = no split! :-)
258 
259  if (TRIM){
260  StringTools::trimScan(str, pos1, pos2, trimChars);
261  appendSubstring(sequence, str, pos1, pos2-pos1);
262  }
263  else
264  appendString(sequence, str);
265 
266  return;
267  }
268  else {
269 
270  // Index of the next separator position
271  std::string::size_type pos = pos1; // "irreversible"
272 
273  while (true) {
274 
275  pos1 = pos;
276  pos = str.find_first_of(separators, pos);
277  if (pos == std::string::npos){
278  pos2 = n;
279  if (TRIM)
280  StringTools::trimScan(str, pos1, pos2, trimChars);
281  appendSubstring(sequence, str, pos1, pos2-pos1);
282  return;
283  }
284  else {
285  pos2 = pos;
286  if (TRIM)
287  StringTools::trimScan(str, pos1, pos2, trimChars);
288  appendSubstring(sequence, str, pos1, pos2-pos1);
289  pos = str.find_first_not_of(trimChars, pos+1); // if separator is space, or other also in trimChars
290  //pos = str.find_last_not_of(trimChars, pos);
291  //++pos; // for the next round
292  }
293 
294  }
295  }

299 template <class T1, class T2, class S>
300 bool StringTools::split2(const std::string & s, T1 & first, T2 & second, const S & separators, const std::string & trimChars){
301 
302  std::size_t i = s.find_first_of(separators);
303 
304  if (i != std::string::npos){ // input of type "key=value" found
305 
306  std::stringstream sstr1;
307  //std::cerr << "eka " << s.substr(0, i) << '\n';
308  if (!trimChars.empty()){
309  sstr1 << StringTools::trim(s.substr(0, i), trimChars);
310  }
311  else
312  sstr1 << s.substr(0, i);
313  sstr1 >> first;
314 
315  std::stringstream sstr2;
316  ++i;
317  if (i<s.size()){
318  //std::cerr << "toka " << s.substr(i) << '\n';
319  if (!trimChars.empty()){
320  sstr2 << StringTools::trim(s.substr(i), trimChars);
321  }
322  else
323  sstr2 << s.substr(i);
324  }
325  sstr2 >> second;
326  return true;
327  }
328  else {
329  std::stringstream sstr;
330  if (trimChars.empty())
331  sstr << s;
332  else
333  sstr << StringTools::trim(s, trimChars);
334  sstr >> first;
335  return false;
336  }
