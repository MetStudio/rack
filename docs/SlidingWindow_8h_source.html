62  void setSlidingMode(bool horzMultiple, bool vertMultiple){
63 
64  modeStr = "??";
65  modeStr.at(0) = horzMultiple?'H':'h';
66  modeStr.at(1) = vertMultiple?'V':'v';
67 
68  updateHorz = horzMultiple ? &SlidingWindow<C,R>::updateHorzMultiple : &SlidingWindow<C,R>::updateHorzSingle;
69  updateVert = vertMultiple ? &SlidingWindow<C,R>::updateVertMultiple : &SlidingWindow<C,R>::updateVertSingle;
70 
71  if (horzMultiple && vertMultiple)
72  fill = & SlidingWindow<C,R>::fillBoth;
73  else if (horzMultiple)
74  fill = & SlidingWindow<C,R>::fillVert; //Horz;
