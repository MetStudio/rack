114  mout.error() << "Projection undefined, cannot create tile" << mout.endl;
115 
116  composite.setCropping(true);
117  add(RackContext::POLAR|RackContext::CURRENT);
118  extract("dw");
119 
120  // "Debugging"
121  if (!composite.isCropping()){
122  mout.warn() << "Composite cropping switch of during op" << mout;
123  mout.error() << "Programming error in parallel comp design" << mout;
124  }
125 
126  composite.setCropping(false);
