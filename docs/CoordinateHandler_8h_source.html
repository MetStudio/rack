188  virtual inline
189  coord_overflow_t handle(int &x,int &y) const {
190 
191  result = 0;
192 
193  if (x < xRange.min)
194  result |= (this->*handleXUnderFlow)(x,y);
195 
196  if (x >= area.width)
197  result |= (this->*handleXOverFlow)(x,y);
198 
199  if (y < yRange.min)
200  result |= (this->*handleYUnderFlow)(x,y);
201 
202  if (y >= area.height)
203  result |= (this->*handleYOverFlow)(x,y);
204 
205  return result;
206 
207  }
