162 template <typename T>
163 typename FlagResolver::value_t FlagResolver::addEntry(drain::Dictionary<key_t,T> &dict, const typename dict_t::key_t & key, value_t i){
164 
165  drain::Logger mout(__FUNCTION__, __FILE__);
166 
167  if (dict.hasKey(key)){
168  mout.info() << key << " already in dict: " << dict << mout.endl;
169  return dict.getValue(key);
170  }
171 
172  if (i==0)
173  i = getFreeBit(dict);
174 
175  if (i>0){
176  dict.add(key, i);
177  }
178  else {
179  mout.warn() << key << " could not get a valid (non-zero) bit flag for dict: " << dict << mout.endl;
180  }
181 
182  return i;
183 };

186 template <typename T>
187 typename FlagResolver::value_t FlagResolver::getValue(const drain::Dictionary<key_t,T> & dict, const std::string & args, char separator){
188 
189  drain::Logger mout(__FUNCTION__, __FILE__);
190 
191  value_t v = 0;
192 
193  std::list<key_t> keys;
194 
195  if (!separator)
196  separator = dict.separator;
197 
198 
199  drain::StringTools::split(args, keys, separator);
200 
201  for (const key_t & key: keys){
202 
203  if (key == ""){
204  continue;
205  }
206  else if (key == "0"){
207  v = 0; // TODO if resets and sets?
208  continue;
209  }
210  // mout.warn() << " '" << *it << "'" << mout.endl;
211 
212  //typename dict_t::const_iterator dit = dict.findByKey(key);
213  typename drain::Dictionary<key_t,T>::const_iterator dit = dict.findByKey(key);
214 
215  if (dit != dict.end()){ // String key match,
216  // Numeric value for an alphabetic key was found
217  v = (v | dit->second);
218  }
219  else {
220  // Numeric value
221  value_t x;
222  std::stringstream sstr(key);
223  sstr >> x; // FIX if empty
224  if (x == 0){
225  // String "0" handled already above
226  mout.error("key '", key, "' not found in Flags, dict: ", dict);
227  //throw std::runtime_error(key, ": key not found in Flags, dict: ", dict);
228  }
229  v = v | x;
230  // Nice to know
231  /*
232  dict_t::const_iterator vit = dictionaryRef.findByValue(v);
233  if (vit != dictionaryRef.end()){
234  std::cout << "(assigned key '" << vit->second << "')\n"; // or vit->first?
235  }
236  */
237  }
238  }
239 
240  return v;
241 }
