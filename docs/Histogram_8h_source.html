423 void Histogram::compute(const T & dst, const std::type_info & type){
424 
425  drain::Logger mout(__FUNCTION__, __FILE__);
426 
427  //const std::type_info & type = dst.getType();
428 
429  const std::size_t s = getSize();
430 
431  //mout.note() << histogram.getSize() << " bins, storage type resolution " << s << " " << mout.endl;
432 
433  if ((s == 256) && (type == typeid(unsigned char))){
434  mout.note() << "direct mode: u char (fast)" << mout.endl;
435  for (typename T::const_iterator it = dst.begin(); it != dst.end(); ++it){
436  this->incrementRaw(static_cast<unsigned short int>(*it));
437  }
438  }
439  else if ((s == 256) && (type == typeid(unsigned short int))){
440  mout.note() << "direct mode: u short (fast)" << mout.endl;
441  for (typename T::const_iterator it = dst.begin(); it != dst.end(); ++it){
442  this->incrementRaw(static_cast<unsigned short int>(*it) >> 8);
443  }
444  }
445  else {
446  mout.note() << "scaled mode (slow)" << mout.endl;
447  this->setScale(dst.getScaling().getMinPhys(), dst.getScaling().getMaxPhys());
448  for (typename T::const_iterator it = dst.begin(); it != dst.end(); ++it){
449  this->increment(*it);
450  }
451  }
452 
453  //mout.note() << "finito" << mout.endl;
454 
455 }
