106 template <class T> // , const CommentReader & commentReader = CommentReader()
107 void FilePng::read(T & image, const std::string & path, int png_transforms ) {
108 
109  drain::Logger mout(getImgLog(), __FUNCTION__, __FILE__);
110 
111  mout.debug() << "path='" << path << "'" << mout.endl;
112 
113 
114  // Try to open the file
115  FILE *fp = fopen(path.c_str(), "rb");
116  if (fp == NULL)
117  throw std::runtime_error(std::string("FilePng: could not open file: ") + path);
118 
119  // For checking magic code (signature)
120  //const unsigned int PNG_BYTES_TO_CHECK=4;
121  const size_t PNG_BYTES_TO_CHECK=4;
122  png_byte buf[PNG_BYTES_TO_CHECK];
123 
124  /* Read in some of the signature bytes */
125  if (fread((void *)buf, size_t(1), PNG_BYTES_TO_CHECK, fp) != PNG_BYTES_TO_CHECK)
126  throw std::runtime_error(std::string("FilePng: suspicious size of file: ") + path);
127 
128  /* Compare the first PNG_BYTES_TO_CHECK bytes of the signature.
129  Return nonzero (true) if they match */
130  if (png_sig_cmp(buf, (png_size_t)0, PNG_BYTES_TO_CHECK) != 0)
131  throw std::runtime_error(std::string("FilePng: not a png file: ")+path);
132 
133 
134 
135  png_structp png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
136  if (!png_ptr){
137  throw std::runtime_error(std::string("FilePng: problem in allocating image memory for: ")+path);
138  }
139 
140  png_infop info_ptr = png_create_info_struct(png_ptr);
141  if (!info_ptr){
142  png_destroy_read_struct(&png_ptr,(png_infopp)NULL, (png_infopp)NULL);
143  throw std::runtime_error(std::string("FilePng: problem in allocating info memory for: ")+path);
144  }
145 
146  /*
147  png_infop end_info = png_create_info_struct(png_ptr);
148  if (!end_info){
149  png_destroy_read_struct(&png_ptr, &info_ptr,(png_infopp)NULL);
150  throw std::runtime_error(std::string("FilePng: problem in allocating end_info memory for: ")+path);
151  }
152  */
153 
154  // This may be unstable. According to the documentation, if one uses the high-level interface png_read_png()
155  // one can only configure it with png_transforms flags (PNG_TRANSFORM_*)
156  png_set_palette_to_rgb(png_ptr);
157 
158  png_init_io(png_ptr, fp);
159  png_set_sig_bytes(png_ptr, PNG_BYTES_TO_CHECK);
160 
162  //if (drain::Debug > 2)
163  mout.debug3() << "reading data" << mout.endl;
164 
165  png_read_png(png_ptr, info_ptr, png_transforms, NULL);
166 
167 
168 
169 
170 
172  mout.debug3() << "reading image comments" << mout.endl;
173  int num_text = 0;
174  png_textp text_ptr = NULL;
175  png_get_text(png_ptr, info_ptr,&text_ptr, &num_text);
176  //mout.debug3() << '\n';
177  for (int i = 0; i < num_text; ++i) {
178  mout << text_ptr[i].key << '=' << text_ptr[i].text << '\n';
179  // ValueReader::scanValue(text_ptr[i].text, image.properties[text_ptr[i].key]);
180  JSONreader::readValue(text_ptr[i].text, image.properties[text_ptr[i].key]);
181  }
182  mout << mout.endl;
183 
184 
185  const unsigned int inputBitDepth = png_get_bit_depth(png_ptr, info_ptr);
186  drain::Type t;
187  switch (inputBitDepth) {
188  case 16:
189  //image.initialize<unsigned short>();
190  t.setType<unsigned short>();
191  break;
192  case 8:
193  t.setType<unsigned char>();
194  break;
195  default:
196  fclose(fp);
197  png_destroy_read_struct(&png_ptr,&info_ptr, (png_infopp)NULL);
198  //png_free_data(png_ptr,info_ptr,PNG_FREE_ALL,-1); // ???
199  throw std::runtime_error(std::string("FilePng: unsupported bit depth in : ")+path);
200  return;
201  }
202 
203  //image.setType(t);
204  mout.debug() << "initialize, type " << image.getType().name() << mout.endl;
205 
207  const unsigned int width = png_get_image_width(png_ptr, info_ptr);
208  const unsigned int height = png_get_image_height(png_ptr, info_ptr);
209  const unsigned int channels = png_get_channels(png_ptr, info_ptr);
210 
211 
212  Geometry g(image.getGeometry());
213 
214  // This test enables read into an alpha channel.
215  if ((channels!=g.channels.getChannelCount()) || (width!=g.area.getWidth()) || (height!=g.area.getHeight())){
216  switch (channels) {
217  case 4:
218  g.set(width,height,3,1);
219  break;
220  case 3:
221  g.setGeometry(width,height,3); // check alpha=0!
222  break;
223  case 2:
224  g.set(width,height,1,1);
225  break;
226  case 1:
227  g.setGeometry(width,height,1); // check alpha=0!
228  break;
229  default:
230  throw std::runtime_error(std::string("FilePng: invalid channel count in : ")+path);
231  }
232  }
233 
234  // Form Image, set target type and geometry. For ImageFrame, compare target type and geometry. If differences, throw exception.
235  image.initialize(t, g);
236 
237  mout.debug3() << "png geometry ok, ";
238  mout << "png channels =" << channels << "\n";
239  mout << "png bit_depth=" << inputBitDepth << "\n";
240  mout << mout.endl;
241 
242  // TODO: use png_get_pCal(.........)
243 #ifdef PNG_pCAL_SUPPORTED___DEFUNCT
244  if (info_ptr->pcal_X0 == info_ptr->pcal_X1){
246  mout.toOStr() << "physical scale supported, but no intensity range, pcalX0=" << info_ptr->pcal_X0 << ", pcalX1=" << info_ptr->pcal_X1 << mout.endl;
247  image.setDefaultLimits();
248  }
249  else {
250  image.setLimits(info_ptr->pcal_X0, info_ptr->pcal_X1);
251  mout.note() << "setting physical scale: " << image << mout.endl;
252  }
253 
254 #endif
255 
256  /*
257  if ((bit_depth!=8) && (bit_depth != 16)){
258  fclose(fp);
259  png_destroy_read_struct(&png_ptr,&info_ptr, (png_infopp)NULL);
260  //png_free_data(png_ptr,info_ptr,PNG_FREE_ALL,-1); // ???
261  throw std::runtime_error(std::string("FilePng: unsupported bit depth in : ")+path);
262  }
263  */
264  const unsigned int targetBitDepth = 8*image.getConf().getElementSize();
265 
266  const bool from8to8 = (inputBitDepth == 8) && (targetBitDepth == 8);
267  const bool from8to16 = (inputBitDepth == 8) && (targetBitDepth == 16);
268 
269  if (from8to8) {
270  mout.debug() << "8-bit input, 8-bit target, easy..." << mout;
271  }
272  else if (from8to16) {
273  mout.note() << "-bit input, 16-bit target, rescaling..." << mout;
274  }
275  else {
276  if ((inputBitDepth == 16) && (targetBitDepth == 16)){
277  mout.debug() << "16-bit input, 16-bit target, ok..." << mout;
278  }
279  else {
280  mout.warn() << inputBitDepth << "-bit input, "<< targetBitDepth << "-bit target, problems ahead?" << mout;
281  }
282  }
283 
284  png_bytep *row_pointers = png_get_rows(png_ptr, info_ptr);
285  png_bytep p;
286  int i0;
287  for (unsigned int j = 0; j < height; ++j) {
288  p = row_pointers[j];
289  for (unsigned int i = 0; i < width; ++i) {
290  for (unsigned int k = 0; k < channels; ++k) {
291  i0 = channels*i + k;
292  if (from8to8) {
293  image.put(i,j,k, p[i0]);
294  }
295  else if (from8to16) {
296  image.put(i,j,k, p[i0]<<8);
297  }
298  else {
299  image.put(i,j,k, (p[i0*2]<<8) + (p[i0*2+1]<<0));
300  }
301  }
302  }
303  }
304 
305  fclose(fp);
306  png_destroy_read_struct(&png_ptr,&info_ptr, (png_infopp)NULL);
307  //png_free_data(png_ptr,info_ptr,PNG_FREE_ALL,-1); // ???
308 
309  //png_destroy_read_struct(&png_ptr,(png_infopp)NULL, (png_infopp)NULL);
310  //png_destroy_info_struct(png_ptr,&info_ptr);
311 
312 
313 }
314 
315 
316 
317 }
318 
319 }
