831  template <class T>
832  void assignContainer(const T & v, bool append=false) {
833 
834  if (!append){
835  if (isStlString())
836  clear();
837  else
838  requestSize(0);
839  }
840 
841  suggestType(typeid(typename T::value_type));
842  // Note: sequence of strings, will request string type...
843 
844  if (isString()){
845  std::stringstream sstr;
846  if (append){
847  toStream(sstr); // outputsepp?
848  if (inputSeparator)
849  sstr << inputSeparator;
850  }
851  StringTools::join(v, sstr, inputSeparator); // implement!
852  assignToString(sstr.str());
853  }
854  else if (caster.typeIsSet()){
855 
856  if (append){
857  for (typename T::const_iterator it = v.begin(); it != v.end(); ++it){
858  appendToElementArray(*it);
859  }
860  }
861  else {
862  requestSize(v.size());
863  typename T::const_iterator it = v.begin();
864  typename T::const_iterator itLast = it;
865  for (size_t i = 0; i<getElementCount(); ++i){
866  if (it == v.end()){
867  if (fillArray)
868  it = itLast; //v.begin(); WHAT if empty?
869  else
870  return;
871  }
872  else {
873  itLast = it;
874  }
875  caster.put(getPtr(i), *it);
876  ++it;
877  }
878  }
879  }
880  else {
881  throw std::runtime_error(std::string(__FILE__) + ": type unset, cannot assign");
