108  if (unicloner.bank.has(functorName)){
109  UnaryFunctor & f = unicloner.getCloned(functorName); // drain::getFunctor(functorStr, ':');
110  if (scale > 0.0)
111  f.setScale(scale, bias);
112  f.setParameters(functorParams);
113  mout.debug3() << " ftor: " << f << mout.endl;
114  // mout.warn() << "cftor: " << (const UnaryFunctor &)f << mout.endl;
115  return f;
116  }
117  else {
118  mout.fail() << "Functor '" << functorName << "' not found. Available functors:" << mout.endl;
119  //std::cerr << unicloner.bank
120  mout.note() << "Available functors:\n" << unicloner.bank << mout.endl;

122  //Sprinter::sequenceToStream(std::cerr, unicloner.bank, layout);
123  mout.warn() << "using default ftor: " << this->functor << mout.endl;
124  return this->functor;
