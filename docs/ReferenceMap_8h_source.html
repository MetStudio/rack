227  template <class T>
228  void copyStruct(const ReferenceMap & m, const T & src, T & dst, extLinkPolicy policy=RESERVE) {
229  //, bool copyValues = true, bool linkExternal = false){
230 
231  Logger mout(__FUNCTION__, __FILE__);
232  long s = sizeof(T); // yes signed
233  //mout.warn() << "experimental, obj.size=" << s << mout.endl;
234 
235  // Clearing is bad, it resets work of base class constructors
236  // if (policy==LINK)
237  // clear();
238 
239  typedef unsigned long addr_t;
240  typedef long addr_diff_t;
241 
242  const addr_t srcAddr = (addr_t)(&src);
243  const addr_t dstAddr = (addr_t)(&dst);
244  for (std::list<std::string>::const_iterator it = m.getKeyList().begin(); it != m.getKeyList().end(); ++it){
245  const std::string & key = *it;
246  const Referencer & srcRef = m[key];
247  addr_t srcVarAddr = (addr_t)srcRef.getPtr();
248  addr_diff_t relativeAddr = srcVarAddr - srcAddr;
249  if ((relativeAddr >= 0) && (relativeAddr < s)){ // INTERNAL
250  Referencer & dstMemberRef = link(key, (void *)(dstAddr + relativeAddr), srcRef.getType(), srcRef.getElementCount());
251  //mout.warn() << "local: " << key << ':' << srcRef.getElementCount() << mout.endl;
252  dstMemberRef.copyFormat(srcRef);
253  dstMemberRef = srcRef; // value
254  }
255  else {
256  //mout.warn() << "external: " << key << mout.endl;
257  switch (policy) {
258  case LINK:
259  link(key, (void *)srcVarAddr, srcRef.getType(), srcRef.getElementCount()).copyFormat(srcRef);
260  break;
261  case RESERVE:
262  reserve(key);
263  //mout.warn() << "reserved: " << key << mout.endl;
264  //mout.warn() << "keyList: " << getKeys() << mout.endl;
265  break;
266  case SKIP:
267  mout.debug() << "skipping external variable: '" << key << '=' << srcRef << "' relative addr=" << relativeAddr << mout.endl;
268  break;
269  case ERROR:
270  mout.error() << "external variable: '" << key << '=' << srcRef << "' relative addr=" << relativeAddr << mout.endl;
271  break;
272  default:
273  mout.warn() << "unknown enum option in handling external variable '" << key << '=' << srcRef << "' relative addr=" << relativeAddr << mout.endl;
274  }
275  }
276  }
277 
278  // Shared properties.
279  //keyList = m.keyList;
280  separator = m.separator;
281  arraySeparator = m.arraySeparator;
282  unitMap = m.unitMap;
283
