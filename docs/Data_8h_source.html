411 void PlainData<DT>::createSimpleQualityData(drain::image::Image & quality, double dataQuality, double nodataQuality, double undetectQuality) const {
412 
413  quality.setPhysicalScale(0.0, 1.0);
414  const drain::image::ImageScaling & scaling = quality.getScaling();
415  const double d = scaling.inv(dataQuality);
416  const double nd = scaling.inv(nodataQuality);
417  const double un = scaling.inv(undetectQuality);
418 
419  quality.setGeometry(data.getWidth(), data.getHeight());
420 
421  Image::iterator it = data.begin();
422  Image::iterator wit = quality.begin();
423  while (it != data.end()){
424  //if ((*it != odim.nodata) && (*it != odim.undetect))
425  if (*it == odim.nodata)
426  *wit = nd;
427  else if (*it == odim.undetect)
428  *wit = un;
429  else
430  *wit = d;
431  ++it;
432  ++wit;
433  }
