183 
184  drain::Logger mout(getImgLog(), "Sampler", __FUNCTION__);

189  std::string format = drain::StringTools::replace(formatStr, "\\n", "\n"); // "\\n", "\n");
190  format = drain::StringTools::replace(format, "\\t", "\t");
191 
192  // Service: associate file keys with data
193  // mout.debug() << "check minus" << mout.endl;
194  std::map<std::string, double> values;
195  static const std::string minusStr("-");
196  for (typename std::map<std::string, D>::const_iterator it = images.begin(); it != images.end(); ++it){
197  const std::string & key = it->first;
198  /*
199  if (it->second.isEmpty()) // not required from D!
200  mout.warn() << key << ": empty image " << it->second << mout.endl;
201  */
202  mout.debug(1) << "referencing: " << key << ',' << minusStr << key << mout.endl;
203  variableMap.reference(key, values[key]=0);
204  variableMap.reference(minusStr+key, values[minusStr+key]=0);
205  }
206  mout.debug() << "variables: " << variableMap << mout.endl;
207 
208 
210  if (!FORMAT){
211  mout.debug() << "constructing default format (all the quantities)" << mout.endl;
212  std::stringstream sstr;
213  const std::list<std::string> & keys = variableMap.getKeyList();
214  char separator = 0;
215  for (std::list<std::string>::const_iterator it=keys.begin(); it!=keys.end(); ++it){
216  if (!it->empty()){
217  if (it->at(0) != '-'){
218  if (separator)
219  sstr << separator;
220  else
221  separator = ',';
222  sstr << '$' << '{' << *it << '}';
223  }
224  }
225  else {
226  mout.warn() << "empty quantity" << mout.endl;
227  }
228  }
229  format = sstr.str();
230  }
231 
232  const int iStep = this->iStep;
233  const int jStep = (this->jStep > 0) ? this->jStep : iStep;
234 
236  if (commentPrefix.empty()){
237  commentChar = 0;
238  }
239  else {
240 
241  if (commentPrefix.size() == 1){
242  commentChar = commentPrefix[0];
243  }
244  else {
245  int i;
246  drain::StringTools::convert(commentPrefix, i);
247  commentChar = i;
248  if (i < 32)
249  mout.warn() << "commentChar bytevalue: " << i << " > commentChar=" << commentChar << mout.endl;
250  else if (i > 128)
251  mout.warn() << "commentChar bytevalue: " << i << " > commentChar=" << commentChar << mout.endl;
252  }
253 
254  //ostr << commentChar << " TEST\n";
255  picker.writeHeader(commentChar, ostr);
256  ostr << commentChar << " size='" << picker.width << 'x' << picker.height << "'\n";
257  const int iN = picker.width/iStep;
258  const int jN = picker.height/jStep;
259  ostr << commentChar << " iStep=" << iStep << "\n";
260  ostr << commentChar << " jStep=" << jStep << "\n";
261  ostr << commentChar << " sampleRows=" << iN << "\n";
262  ostr << commentChar << " sampleCols=" << jN << "\n";
263  ostr << commentChar << " samples=" << (iN*jN) << "\n";
264  if (!formatStr.empty())
265  ostr << commentChar << " format='" << formatStr << "'\n"; // formatStr instead of format, to save double slash \\n \\t
266  else
267  ostr << commentChar << " format='" << format << "'\n";
268  }
269 
270  // Note: supports leading minus sign
271  drain::StringMapper formatter("-?[a-zA-Z0-9_]+"); // WAS: "-?[a-zA-Z0-9_]+" with odd "-?"
272  formatter.parse(format);
273  //mout.warn() << "formatter " << formatter << mout.endl;
274 
275 
276  int iStart = this->iRange.min; // this->iStart;
277  if (iStart < 0)
278  iStart = iStep/2;
279 
280  int jStart = this->jRange.min; //jStart;
281  if (jStart < 0)
282  jStart = jStep/2;
283 
284  int iEnd = this->iRange.max; // iEnd;
285  if (iEnd < iStart)
286  iEnd = picker.width-1;
287 
288  int jEnd = this->jRange.max; // jEnd;
289  if (jEnd < jStart)
290  jEnd = picker.height-1;
291 
292 
293  // Main loop: traverse image area with (i,j)
294  // But no referencing here because some applications might prefer derived variables like
295  // j2=(height-1-j) or geographical coords.
296  // => Set them in SamplePicker.
297  double x;
298  bool dataOk;
299  for (int j = jStart; j<jEnd; j+=jStep){
300  for (int i = iStart; i<iEnd; i+=iStep){
301 
302  picker.setPosition(i, j);
303 
304  // SLOW, but works... // TODO speedup with iterator
305  dataOk = true;
306  for (typename std::map<std::string, D>::const_iterator it = images.begin(); it != images.end(); ++it){
307  const std::string & quantity = it->first;
308  const D & data = it->second;
309  if (!picker.getValue(data, x)){
310  dataOk = false;
311  }
312  //else if (x != 0) std::cerr << x << '\t';
313  values[quantity] = x;
314  values[minusStr+quantity] = -x;
315  }
316 
317  if (dataOk || (skipVoid==0)){
318  formatter.toStream(ostr, variableMap, true);
319  if ((!dataOk) && (commentChar))
320  ostr << ' ' << commentChar << voidMarker;
321  ostr << '\n';
322  }
323  }
324  // formatter.expand(map, true);
325  }
326  mout.debug() << "last values: " << variableMap << mout.endl;
327  //mout.warn() << "formatter " << formatter << mout.endl;
328 
329  }
330 
331 
332 };
