126 void somToImage(const Som< T > & som, drain::image::Image & image, int width=0, int height=0, const std::vector<int> & channels = std::vector<int>()){
127 
128  const int depth = som.map[0][0].size();
129 
130  std::vector<int> defaultChannels;
131 
132  const std::vector<int> & applyChannels = channels.empty() ? defaultChannels : channels;
133 
134  if (channels.empty()){
135  defaultChannels.resize(depth);
136  for (int k = 0; k < depth; ++k) {
137  defaultChannels[k] = k;
138  }
139  }
140 
141  const int n = applyChannels.size();
142 
143  if ((n != 1) && (n != 3)){
144  std::cerr << "Warning: som2PNM: dimension not 1 (gray) or 3 (rgb) " << std::endl;
145  return;
146  }
147 
148  const int w = (width > 0) ? width : som.getWidth();
149  const int h = (height > 0) ? height : som.getHeight();
150 
151  image.setGeometry(w, h, n);
152  /*
153  cout << "Writing image of " << image.getGeometry() << endl;
154  for (int k = 0; k < n; ++k)
155  cout << applyChannels[k] << ' ';
156  cout << endl;
157  */
158 
159 
160  int iMap, jMap;
161  for (int j = 0; j < h; ++j) {
162  jMap = (j * som.getHeight()) / h;
163  for (int i = 0; i < w; ++i) {
164  iMap = (i * som.getWidth()) / w;
165  for (int k = 0; k < n; ++k) {
166  image.put(i,j,k, som.map[jMap][iMap][applyChannels[k]]);
