154  drain::Logger mout(getImgLog(), getName(), __FUNCTION__);

156  const size_t w = src.getWidth();
157  const size_t h = src.getHeight();

159  const src_t minRaw = src.getScaling().inv(min);
160  const src_t maxRaw = (max == std::numeric_limits<double>::max()) ? src.getEncoding().getTypeMax<src_t>() : src.getScaling().inv(max);
161 
162  if (minRaw <= src.getEncoding().getTypeMin<src_t>()){
163  mout.warn() << "min value=" << (double)minRaw << " less or smaller than storage type min=" << src.getEncoding().getTypeMin<src_t>() << mout.endl;

168  mout.debug(1) << "dst: " << dst << mout.endl;
169

171  SizeProber sizeProber(src, dst);

173  sizeProber.conf.anchorMax = maxRaw;
174  mout.debug(1) << "areaProber:" << sizeProber << mout.endl;
175 
176  FillProber floodFill(src, dst);
177  floodFill.conf.anchorMin = minRaw;
178  floodFill.conf.anchorMax = maxRaw;
179  mout.debug(1) << "floodFill: " << floodFill << mout.endl;

187 
188  mout.debug() << "Scale: " << scale << mout.endl;
189 
190  const UnaryFunctor & ftor = getFunctor(scale);
