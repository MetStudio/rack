248  void probe8(int i, int j) {
249 
250  // Note: coordHandler applied by checkPos
251  if (!handler.validate(i, j))
252  return; // false;
253 
254  if (isVisited(i,j))
255  return; // false;
256 
257  if (!isValidSegment(i, j))
258  return; // false;
259 
260  visit(i,j); // mark & update
261 
263  int i2 = i;
264  while (moveHorz(i2,j, i2+1,j)){
265  ++i2;
266  visit(i2, j);
267  }
268  // Now i2 is the maximum valid i coordinate.
269 
271  while (moveHorz(i,j, i-1,j)){
272  --i;
273  visit(i, j);
274  }
275  // Now i is the minimum valid i coordinate.
276 
278  while (i <= i2){
279  if (isValidMove(i,j, i,j-1))
280  probe8(i,j-1);
281  if (isValidMove(i,j, i,j+1))
282  probe8(i,j+1);
283  ++i;
284  }
285
