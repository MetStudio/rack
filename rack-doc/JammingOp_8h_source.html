159  inline
160  static
161  double derivativeDBZ(const PolarODIM &odimIn, const Image &src, const int & i, const int & j, const int span = 10){
162  const int iLower = std::max(i-span, 0);
163  const double zLower = src.get<double>(iLower,j);
164  if ((zLower != odimIn.nodata)&&(zLower != odimIn.undetect)){
165  const int iUpper = std::min(i+span, static_cast<int>(src.getWidth())-1);
166  const double zUpper = src.get<double>(iUpper,j);
167  if ((zUpper != odimIn.nodata)&&(zUpper != odimIn.undetect)){
168  return (zUpper - zLower)/(odimIn.rscale*static_cast<double>(iUpper-iLower));
169  }
170  }
171  return std::numeric_limits<double>::max();
172  }
