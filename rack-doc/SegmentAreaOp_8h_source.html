153 
154  drain::Logger mout(getImgLog(), getName(), __FUNCTION__);
155 
156  const size_t w = src.getWidth();
157  const size_t h = src.getHeight();
158 
159  const src_t minRaw = src.getScaling().inv(min);
160  const src_t maxRaw = (max == std::numeric_limits<double>::max()) ? src.getMax<src_t>() : src.getScaling().inv(max);
161 
162  if (minRaw <= src.getMin<src_t>()){
163  mout.warn() << "min value=" << (double)minRaw << " less or smaller than storage type min=" << src.getMin<src_t>() << mout.endl;
164  }
165 
166  mout.debug() << "raw range: " << (double)minRaw << '-' << (double)maxRaw << mout.endl;
167  mout.debug(1) << "src: " << src << mout.endl;
168  mout.debug(1) << "dst: " << dst << mout.endl;
169 
170  SegmentProber<S,D> floodFill(src, dst);
171 
172  //T floodFill(src, dst);
173 
174  //const size_t dMax = dst.getMax<size_t>();
175  // const typename T::dst_t dMax = dst.getMax<typename T::dst_t>();
176  // mout.warn() << "dMax " << (double)dMax << mout.endl;
177  const double scale = drain::Type::call<drain::typeIsSmallInt>(dst.getType()) ? dst.getMax<double>() : 1.0;
178 
179  const UnaryFunctor & ftor = getFunctor(scale);
180  //const UnaryFunctor & ftor = getFunctor(dst.getMax<T::dst_t>());
181 
182  mout.debug(1) << "Functor: " << ftor.getName() << '=' << ftor.getParameters() << mout.endl;
183  /*
184  mout.warn() << "Functor: 1 =>" << ftor(1.0) << mout.endl;
185  mout.warn() << "Functor: 100 =>" << ftor(100.0) << mout.endl;
186  mout.warn() << "Functor: 10000 =>" << ftor(10000.0) << mout.endl;
187  */
188  //mout.debug(1);
189  mout << mout.endl;
190 
191  typedef drain::typeLimiter<dst_t> Limiter;
192  typename Limiter::value_t limiter = dst.getLimiter<dst_t>();
193 
194  size_t sizeMapped;
195  for (size_t i=0; i<w; i++){
196  for (size_t j=0; j<h; j++){
197  //if ((drain::Debug > 5) && ((i&15)==0) && ((j&15)==0)) std::cerr << i << ',' << j << "\n";

199 
200  // STAGE 1: detect size.
201  floodFill.probe(i,j,1, minRaw, maxRaw); // painting with '1' does not disturb dst
202 
203  if (floodFill.size > 0){

207  sizeMapped = limiter(ftor(floodFill.size));
208  if (sizeMapped == 0)

211  /*

213  mout << " size=" << floodFill.size << " => "<< sizeMapped << mout.endl;
214  */
215 
216  floodFill.probe(i,j,sizeMapped, minRaw, maxRaw);
217  //floodFill.fill(i,j,size&254,min,max);
218  //std::cerr << "filled\n";
219  }
220  }
